local dragging = false
local dragInput
local dragStart
local startPos

button.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = button.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

button.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if dragging and input == dragInput then
		local delta = input.Position - dragStart
		button.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

local function createSlider(text)
	local frame = Instance.new("Frame")
	frame.Size = UDim2.new(1, -30, 0, 50)
	frame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
	frame.Parent = menu

	local s = Instance.new("UIStroke", frame)
	s.Color = Color3.fromRGB(150, 150, 150)

	local c = Instance.new("UICorner", frame)
	c.CornerRadius = UDim.new(0, 8)

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(0.5, 0, 0.4, 0)
	label.Position = UDim2.new(0, 10, 0.05, 0)
	label.Text = text
	label.TextColor3 = Color3.new(1, 1, 1)
	label.TextScaled = true
	label.BackgroundTransparency = 1
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.Parent = frame

	local sliderBg = Instance.new("Frame")
	sliderBg.Size = UDim2.new(0.5, -15, 0, 8)
	sliderBg.Position = UDim2.new(0, 10, 0.5, -4)
	sliderBg.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
	sliderBg.Parent = frame

	local sliderCorner = Instance.new("UICorner", sliderBg)
	sliderCorner.CornerRadius = UDim.new(1, 0)

	local sliderKnob = Instance.new("Frame")
	sliderKnob.Size = UDim2.new(0, 16, 0, 16)
	sliderKnob.Position = UDim2.new(0, 0, 0.5, -8)
	sliderKnob.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
	sliderKnob.Parent = sliderBg

	local knobCorner = Instance.new("UICorner", sliderKnob)
	knobCorner.CornerRadius = UDim.new(1, 0)

	local input = Instance.new("TextBox")
	input.Size = UDim2.new(0.35, -10, 0.5, 0)
	input.Position = UDim2.new(1, -50, 0.25, 0)
	input.Text = "10"
	input.TextScaled = true
	input.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
	input.TextColor3 = Color3.new(1, 1, 1)
	input.Parent = frame

	local inputCorner = Instance.new("UICorner", input)
	inputCorner.CornerRadius = UDim.new(0, 4)

	local currentValue = 10

	local function updateSlider(val)
		currentValue = math.max(10, math.min(100, val))
		local percent = (currentValue - 10) / (100 - 10)
		sliderKnob.Position = UDim2.new(percent, -8, 0.5, -8)
		input.Text = tostring(currentValue)
	end

	input.Changed:Connect(function()
		local val = tonumber(input.Text) or 10
		updateSlider(val)
	end)

	input.FocusLost:Connect(function()
		updateSlider(tonumber(input.Text) or currentValue)
	end)

	local draggingSlider = false

	sliderKnob.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			draggingSlider = true
		end
	end)

	sliderKnob.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			draggingSlider = false
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if draggingSlider and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			local sliderSize = sliderBg.AbsoluteSize.X
			local sliderPos = sliderBg.AbsolutePosition.X
			local mousePos = input.Position.X
			local percent = math.max(0, math.min(1, (mousePos - sliderPos) / sliderSize))
			local val = 10 + (percent * (100 - 10))
			updateSlider(val)
		end
	end)

	local function getValue()
		return currentValue
	end

	return frame, getValue
end

local function toggleLogic(toggle, knob)
	local state = false
	toggle.MouseButton1Click:Connect(function()
		state = not state
		if state then
			knob:TweenPosition(UDim2.new(1, -22, 0.5, -10), "Out", "Quad", 0.15, true)
			toggle.BackgroundColor3 = Color3.fromRGB(0, 170, 0)
		else
			knob:TweenPosition(UDim2.new(0, 2, 0.5, -10), "Out", "Quad", 0.15, true)
			toggle.BackgroundColor3 = Color3.fromRGB(70, 70, 70)
		end
	end)
	return function() return state end
end

local auraState = toggleLogic(auraToggle, auraKnob)
local specState = toggleLogic(specToggle, specKnob)
local tpState = toggleLogic(tpToggle, tpKnob)
local distState = toggleLogic(distToggle, distKnob)
local crosshairState = toggleLogic(crosshairToggle, crosshairKnob)
local aimbotState = toggleLogic(aimbotToggle, aimbotKnob)

local function fillList(list, box)
	list:ClearAllChildren()
	local layout = Instance.new("UIListLayout", list)
	for _, p in ipairs(Players:GetPlayers()) do
		if p ~= player then
			local b = Instance.new("TextButton")
			b.Size = UDim2.new(1, 0, 0, 30)
			b.Text = p.Name
			b.TextScaled = true
			b.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
			b.TextColor3 = Color3.new(1, 1, 1)
			b.Parent = list
			b.MouseButton1Click:Connect(function()
				box.Text = p.Name
				list.Visible = false
				arrowSpec.Text = "→"
				arrowTp.Text = "→"
			end)
		end
	end
end

arrowSpec.MouseButton1Click:Connect(function()
	listSpec.Visible = not listSpec.Visible
	arrowSpec.Text = listSpec.Visible and "↓" or "→"
	fillList(listSpec, boxSpec)
end)

arrowTp.MouseButton1Click:Connect(function()
	listTp.Visible = not listTp.Visible
	arrowTp.Text = listTp.Visible and "↓" or "→"
	fillList(listTp, boxTp)
end)

RunService.RenderStepped:Connect(function()
	if auraState() then
		for _, p in ipairs(Players:GetPlayers()) do
			if p ~= player and p.Character then
				if not auraHighlights[p] then
					local hl = Instance.new("Highlight")
					hl.FillColor = Color3.fromRGB(255, 0, 0)
					hl.OutlineColor = Color3.fromRGB(255, 0, 0)
					hl.Parent = p.Character
					auraHighlights[p] = hl
				end
			end
		end
	else
		for _, hl in pairs(auraHighlights) do
			if hl then hl:Destroy() end
		end
		auraHighlights = {}
	end

	if specState() and boxSpec.Text ~= "" then
		local target = Players:FindFirstChild(boxSpec.Text)
		if target and target.Character and target.Character:FindFirstChild("Humanoid") then
			camera.CameraSubject = target.Character:FindFirstChild("Humanoid")
		end
	else
		if player.Character and player.Character:FindFirstChild("Humanoid") then
			camera.CameraSubject = player.Character:FindFirstChild("Humanoid")
		end
	end

	if tpState() and boxTp.Text ~= "" then
		local target = Players:FindFirstChild(boxTp.Text)
		if target and target.Character and player.Character then
			local hrp = target.Character:FindFirstChild("HumanoidRootPart")
			local my = player.Character:FindFirstChild("HumanoidRootPart")
			if hrp and my then
				my.CFrame = hrp.CFrame
			end
		end
	end

	if distState() and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
		local distLabel = playerGui:FindFirstChild("DistanceLabel")
		if not distLabel then
			distLabel = Instance.new("TextLabel")
			distLabel.Name = "DistanceLabel"
			distLabel.Size = UDim2.new(0, 200, 0, 300)
			distLabel.Position = UDim2.new(0, 10, 0.5, -150)
			distLabel.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
			distLabel.TextColor3 = Color3.new(1, 1, 1)
			distLabel.TextScaled = false
			distLabel.TextSize = 12
			distLabel.TextWrapped = true
			distLabel.Font = Enum.Font.GothamMonospace
			distLabel.BorderSizePixel = 0
			distLabel.Parent = playerGui

			local distStroke = Instance.new("UIStroke", distLabel)
			distStroke.Color = Color3.fromRGB(150, 150, 150)
			distStroke.Thickness = 1

			local distCorner = Instance.new("UICorner", distLabel)
			distCorner.CornerRadius = UDim.new(0, 8)
		end

		local distText = ""
		local playerPos = player.Character.HumanoidRootPart.Position
		for _, p in ipairs(Players:GetPlayers()) do
			if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
				local distance = (playerPos - p.Character.HumanoidRootPart.Position).Magnitude
				distText = distText .. p.Name .. " - Distance: " .. math.floor(distance) .. "\n"
			end
		end
		distLabel.Text = distText
	else
		local distLabel = playerGui:FindFirstChild("DistanceLabel")
		if distLabel then
			distLabel:Destroy()
		end
	end

	if crosshairState() then
		crosshairCircle.Visible = true
		local size = getCrosshairSize()
		crosshairCircle.Size = UDim2.new(0, size, 0, size)
		crosshairCircle.Position = UDim2.new(0.5, -size / 2, 0.5, -size / 2)
	else
		crosshairCircle.Visible = false
	end

	if aimbotState() then
		if player.Character and player.Character:FindFirstChild("Humanoid") then
			local closestPlayer = nil
			local closestDistance = math.huge

			for _, p in ipairs(Players:GetPlayers()) do
				if p ~= player and p.Character and p.Character:FindFirstChild("Humanoid") and p.Character:FindFirstChild("Head") then
					if player.Character:FindFirstChild("Head") then
						local distance = (p.Character.Head.Position - player.Character.Head.Position).Magnitude
						if distance < closestDistance then
							closestDistance = distance
							closestPlayer = p
						end
					end
				end
			end

			if closestPlayer and closestPlayer.Character and closestPlayer.Character:FindFirstChild("Head") then
				local targetPosition = closestPlayer.Character.Head.Position
				local cameraDirection = (targetPosition - camera.CFrame.Position).Unit
				camera.CFrame = CFrame.new(camera.CFrame.Position, camera.CFrame.Position + cameraDirection)
			end
		end
	end
end)

Players.PlayerAdded:Connect(function()
	if auraState() then
		local newPlayer = Players:GetChildren()[#Players:GetChildren()]
		if newPlayer and newPlayer ~= player and newPlayer.Character then
			local hl = Instance.new("Highlight")
			hl.FillColor = Color3.fromRGB(255, 0, 0)
			hl.OutlineColor = Color3.fromRGB(255, 0, 0)
			hl.Parent = newPlayer.Character
			auraHighlights[newPlayer] = hl
		end
	end
end)

Players.PlayerRemoving:Connect(function(p)
	if auraHighlights[p] then
		auraHighlights[p]:Destroy()
		auraHighlights[p] = nil
	end
end)
