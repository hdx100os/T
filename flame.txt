local dragging = false
local dragInput
local dragStart
local startPos

local function update(input)
	local delta = input.Position - dragStart
	button.Position = UDim2.new(
		startPos.X.Scale,
		startPos.X.Offset + delta.X,
		startPos.Y.Scale,
		startPos.Y.Offset + delta.Y
	)
end

button.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = button.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

button.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)

	local containerCorner = Instance.new("UICorner")

	containerCorner.Parent = container

	local textLabel = Instance.new("TextLabel")
	textLabel.Size = UDim2.new(0.65, -10, 1, 0)
	textLabel.Position = UDim2.new(0, 8, 0, 0)
	textLabel.BackgroundTransparency = 1

	textLabel.Parent = container

	local selectorBox = Instance.new("Frame")
	selectorBox.Size = UDim2.new(0.25, 0, 1, 0)

	searchInput.TextScaled = true

	expandButton.Size = UDim2.new(0, 25, 1, 0)

	expandButton.BackgroundTransparency = 1

	expandButton.TextColor3 = Color3.fromRGB(255, 255, 255)
	expandButton.TextScaled = true
	expandButton.Font = Enum.Font.GothamBold
	expandButton.BorderSizePixel = 0
	expandButton.Parent = selectorBox

	local listFrame = Instance.new("Frame")
	listFrame.Size = UDim2.new(1, -16, 0, 0)
	listFrame.Position = UDim2.new(0, 8, 1, 8)
	listFrame.Visible = false
	listFrame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
	listFrame.BorderSizePixel = 0
	listFrame.Parent = container
	listFrame.ClipsDescendants = true

	local listStroke = Instance.new("UIStroke")
	listStroke.Color = Color3.fromRGB(128, 128, 128)
	listStroke.Thickness = 1
	listStroke.Parent = listFrame

	local listCorner = Instance.new("UICorner")
	listCorner.CornerRadius = UDim.new(0, 6)
	listCorner.Parent = listFrame

	local listLayout = Instance.new("UIListLayout")
	listLayout.Padding = UDim.new(0, 5)
	listLayout.SortOrder = Enum.SortOrder.LayoutOrder
	listLayout.Parent = listFrame

	listPadding.PaddingLeft = UDim.new(0, 5)
	listPadding.PaddingRight = UDim.new(0, 5)
	listPadding.PaddingTop = UDim.new(0, 5)
	listPadding.PaddingBottom = UDim.new(0, 5)
	listPadding.Parent = listFrame

	local expanded = false
	local selectedPlayer = nil

	local function toggleList()
		expanded = not expanded
		if expanded then
			expandButton.Text = "↓"
			listFrame.Visible = true
		else
			expandButton.Text = "→"
			listFrame.Visible = false
		end
	end

	expandButton.MouseButton1Click:Connect(function()
		toggleList()
	end)

	local function refreshList(filterText)
		listFrame:ClearAllChildren()
		listLayout = Instance.new("UIListLayout")
		listLayout.Padding = UDim.new(0, 5)
		listLayout.SortOrder = Enum.SortOrder.LayoutOrder
		listLayout.Parent = listFrame

		listPadding = Instance.new("UIPadding")
		listPadding.PaddingLeft = UDim.new(0, 5)
		listPadding.PaddingRight = UDim.new(0, 5)
		listPadding.PaddingTop = UDim.new(0, 5)
		listPadding.PaddingBottom = UDim.new(0, 5)
		listPadding.Parent = listFrame

		for _, plr in ipairs(Players:GetPlayers()) do
			if plr ~= player then
				if filterText == "" or string.find(plr.Name:lower(), filterText:lower()) then
					local playerBtn = Instance.new("TextButton")
					playerBtn.Size = UDim2.new(1, 0, 0, 25)
					playerBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
					playerBtn.Text = plr.Name
					playerBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
					playerBtn.TextScaled = true
					playerBtn.Font = Enum.Font.Gotham
					playerBtn.BorderSizePixel = 0
					playerBtn.Parent = listFrame

					local btnCorner = Instance.new("UICorner")
					btnCorner.CornerRadius = UDim.new(0, 5)
					btnCorner.Parent = playerBtn

					playerBtn.MouseButton1Click:Connect(function()
						selectedPlayer = plr
						searchInput.Text = plr.Name
						toggleList()
						layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
							scrollFrame.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + 16)
							local newMenuSize = math.min(500, layout.AbsoluteContentSize.Y + 55)
							menu.Size = UDim2.new(0, 280, 0, newMenuSize)
						end)
					end)
				end
			end
		end

		listLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
			listFrame.Size = UDim2.new(1, -16, 0, math.min(150, listLayout.AbsoluteContentSize.Y + 10))
			layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
				scrollFrame.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + 16)
				local newMenuSize = math.min(500, layout.AbsoluteContentSize.Y + 55)
				menu.Size = UDim2.new(0, 280, 0, newMenuSize)
			end)
		end)
	end

	searchInput.Changed:Connect(function()
		refreshList(searchInput.Text)
	end)

	searchInput.FocusLost:Connect(function(enterPressed)
		if enterPressed then
			for _, plr in ipairs(Players:GetPlayers()) do
				if plr.Name:lower() == searchInput.Text:lower() then
					selectedPlayer = plr
					break
				end
			end
		end
	end)

	local function getSelectedPlayer()
		return selectedPlayer
	end

	return container, getSelectedPlayer
end

local auraContainer, getAuraState = createToggleButton("ESP/Aura", "Highlight all players")
local spectateContainer, getSpectateState = createToggleButton("Spectate", "View player camera")
local spectateSelectContainer, getSpectatePlayer = createPlayerSelector("Spectate Player")
local teleportContainer, getTeleportState = createToggleButton("Teleport", "Teleport to player")
local teleportSelectContainer, getTeleportPlayer = createPlayerSelector("Teleport Player")
local distanceContainer, getDistanceState = createToggleButton("Distance Tracker", "Show player distances")

local highlights = {}

RunService.RenderStepped:Connect(function()
	if getAuraState() then
		for _, plr in ipairs(Players:GetPlayers()) do
			if plr ~= player and plr.Character then
				if not highlights[plr] then
					local hl = Instance.new("Highlight")
					hl.FillColor = Color3.fromRGB(255, 0, 0)
					hl.OutlineColor = Color3.fromRGB(255, 0, 0)
					hl.Parent = plr.Character
					highlights[plr] = hl
				end
			end
		end
	else
		for _, hl in pairs(highlights) do
			if hl then hl:Destroy() end
		end
		highlights = {}
	end

	if getSpectateState() then
		local target = getSpectatePlayer()
		if target and target.Character then
			local hum = target.Character:FindFirstChild("Humanoid")
			if hum then
				camera.CameraSubject = hum
			end
		end
	else
		if player.Character and player.Character:FindFirstChild("Humanoid") then
			camera.CameraSubject = player.Character:FindFirstChild("Humanoid")
		end
	end

	if getTeleportState() then
		local target = getTeleportPlayer()
		if target and target.Character and player.Character then
			local hrp = target.Character:FindFirstChild("HumanoidRootPart")
			local myhrp = player.Character:FindFirstChild("HumanoidRootPart")
			if hrp and myhrp then
				myhrp.CFrame = hrp.CFrame
			end
		end
	end

	if getDistanceState() then
		local distanceLabel = playerGui:FindFirstChild("DistanceLabel")
		if not distanceLabel then
			distanceLabel = Instance.new("TextLabel")
			distanceLabel.Name = "DistanceLabel"
			distanceLabel.Size = UDim2.new(0, 300, 0, 400)
			distanceLabel.Position = UDim2.new(0, 10, 0.5, -200)
			distanceLabel.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
			distanceLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
			distanceLabel.TextScaled = false
			distanceLabel.TextSize = 14
			distanceLabel.TextWrapped = true
			distanceLabel.Font = Enum.Font.Gotham
			distanceLabel.BorderSizePixel = 0
			distanceLabel.Parent = playerGui

			local distanceStroke = Instance.new("UIStroke")
			distanceStroke.Color = Color3.fromRGB(128, 128, 128)
			distanceStroke.Thickness = 2
			distanceStroke.Parent = distanceLabel

			local distanceCorner = Instance.new("UICorner")
			distanceCorner.CornerRadius = UDim.new(0, 10)
			distanceCorner.Parent = distanceLabel
		end

		local distanceText = ""
		if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
			local playerPos = player.Character.HumanoidRootPart.Position
			for _, plr in ipairs(Players:GetPlayers()) do
				if plr ~= player and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
					local distance = (playerPos - plr.Character.HumanoidRootPart.Position).Magnitude
					distanceText = distanceText .. plr.Name .. " - Distance: " .. math.floor(distance) .. "\n"
				end
			end
		end

		distanceLabel.Text = distanceText
	else
		local distanceLabel = playerGui:FindFirstChild("DistanceLabel")
		if distanceLabel then
			distanceLabel:Destroy()
		end
	end
end)

	for _, selector in ipairs(scrollFrame:GetChildren()) do

			local searchInput = selector:FindFirstChild("SearchInput")

end)
